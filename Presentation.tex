\begin{frame}{Introduction}

\textbf{Data}
\begin{itemize}
\item Spatial measurements of tunneling current $\mathrm{I}(x,y)$
\item Observations exhibit significant stochastic variability
\end{itemize}
\break
\vspace{0.3cm}

\textbf{Main Objective}
\begin{itemize}
\item Identify the appropriate \textbf{statistical noise model}
\end{itemize}

\break
\vspace{0.3cm}

\textbf{Approach (Overview)}
\begin{itemize}
\item Model spatial structure and compare competing noise assumptions
\item Gaussian (continuous) vs Poisson (count-based)
\end{itemize}

\end{frame}


\begin{frame}{Modeling Framework}

\textbf{Spatial Structure}
\begin{itemize}

\item Flexible approximation of unknown spatial structure

\begin{equation}
    f(x,y) = \beta_0 + \beta_1 x + \beta_2 y + \beta_3 x^2 + \beta_4 xy + \beta_5 y^2 + \cdots  = \sum_{i+j \le d} \beta_{ij}, x^i y^j
\end{equation}

\item Model complexity controlled via degree $d$
\end{itemize}

\break
\vspace{0.1cm}

\textbf{Two Noise Formulations}

\begin{itemize}
\item \textbf{Gaussian Model (log-space)}
\begin{equation}
    z = \log(I) = f(x,y) + \epsilon, \quad \epsilon \sim \mathcal{N}(0,\sigma^2)    
\end{equation}


\break
\item \textbf{Poisson Model}
\begin{equation}
    I \sim \text{Poisson}(\lambda(x,y)), \quad \log \lambda = f(x,y)
\end{equation}


\end{itemize}

\break
\vspace{0.1cm}

\textbf{Key Idea}
\begin{itemize}
\item Same spatial model, different noise assumptions
\end{itemize}

\end{frame}


\begin{frame}{Gaussian Model: Formulation and Estimation}

\begin{columns}

% -------- LEFT SIDE --------
\column{0.55\textwidth}

\textbf{Model Assumption}
\begin{equation*}
z_i = \log(I_i) = f(x_i,y_i) + \epsilon_i, 
\quad \epsilon_i \sim \mathcal{N}(0,\sigma^2)
\end{equation*}
\textbf{Likelihood Function}
\begin{equation*}
L(\beta,\sigma^2) = \prod_{i=1}^{N}
\frac{1}{\sqrt{2\pi\sigma^2}}
\exp\left(-\frac{(z_i - f(x_i,y_i))^2}{2\sigma^2}\right)
\end{equation*}

\begin{equation*}
\log L = -\frac{N}{2}\log(2\pi\sigma^2)
- \frac{1}{2\sigma^2} \sum_{i=1}^{N} (z_i - f(x_i,y_i))^2
\end{equation*}



% -------- RIGHT SIDE --------
\column{0.45\textwidth}
\centering
\includegraphics[width=0.9\textwidth]{fig1.png}


\end{columns}

\textbf{MLE Estimation}
\begin{equation*}
\hat{\beta} = \arg\min_{\beta} \mathrm{RSS}, 
\quad \mathrm{RSS} = \sum_{i=1}^{N} (z_i - f(x_i,y_i))^2, 
\quad\hat{\sigma}^2 = \frac{\mathrm{RSS}}{N}
\end{equation*}



\end{frame}


\begin{frame}{Gaussian Model: Inference and Validation}

\begin{columns}

\column{0.55\textwidth}

\textbf{Parameter Uncertainty}
\begin{equation*}
\mathrm{Cov}(\hat{\beta}) = \sigma^2 (X^T X)^{-1}
\end{equation*}

\begin{equation*}
t_j = \frac{\hat{\beta}_j}{\mathrm{SE}(\hat{\beta}_j)}
\end{equation*}

\vspace{0.2cm}

\textbf{Goodness of Fit}
\begin{equation*}
\chi^2 = \sum_{i=1}^{N} \frac{(z_i - \hat{z}_i)^2}{\sigma^2}
\quad
\chi^2_{\text{red}} = \frac{\chi^2}{N - p}
\end{equation*}

\vspace{0.2cm}

\column{0.45\textwidth}

\vfill
\centering
\includegraphics[width=0.9\textwidth]{fig2.png}
\vfill

\end{columns}

\textbf{Model Selection}
\begin{equation*}
\mathrm{AIC} = N \log\left(\frac{\mathrm{RSS}}{N}\right) + 2k
\quad
\mathrm{BIC} = N \log\left(\frac{\mathrm{RSS}}{N}\right) + k \log N
\end{equation*}

\begin{equation*}
F = \frac{(\mathrm{RSS}_d - \mathrm{RSS}_{d+1})/k}
{\mathrm{RSS}_{d+1}/(N - p)}
\end{equation*}

\end{frame}








\begin{frame}{AIC \& BIC (Model Selection)}
\begin{equation}
    \text{AIC} = 2k - 2\ln L
\end{equation}
\begin{equation}
    \text{BIC} = k \ln n - 2\ln L
\end{equation}

\begin{itemize}
    \item Balance model fit vs.\ complexity
    \item Lower values indicate a better model; comparisons are made using differences.
\end{itemize}

\begin{frame}{Reduced Chi-Square ($\chi^2_\nu$)}
\begin{columns}

% -------- LEFT SIDE --------
\column{0.55\textwidth}
\textbf{Chi-Square}
\begin{equation*}
\chi^2 = \sum_{i=1}^{n} \frac{(y_i - y_i^{\text{model}})^2}{\sigma_i^2}
\end{equation*}

\begin{itemize}
    \item Measures goodness of fit
\end{itemize}
\textbf{Reduced Chi-Square}
\begin{equation*}
\chi^2_\nu = \frac{\chi^2}{\nu}, \quad \nu = n - k
\end{equation*}

\begin{itemize}
    \item $\nu = n - k$
    \item $\chi^2_\nu \approx 1$: good \quad $\gg 1$: poor \quad $\ll 1$: overfit
\end{itemize}

\begin{frame}{RSS, Variance and Deviance}
\begin{columns}

% -------- LEFT SIDE --------
\column{0.55\textwidth}

\textbf{Gaussian: RSS \& Variance}
\begin{itemize}
    \item $\text{RSS} = \sum (y_i - y_i^{\text{model}})^2$ measures total squared error
    \item Variance: $\sigma^2 \approx \frac{\text{RSS}}{n - k}$ estimates noise level
    \item Lower RSS / variance $\Rightarrow$ better fit
\end{itemize}
\vspace{0.2cm}

\textbf{Poisson: Deviance}
\begin{itemize}
    \item Measures goodness of fit relative to a saturated model
    \item $D = 2 \sum \left[y_i \ln\left(\frac{y_i}{\hat{y}_i}\right) - (y_i - \hat{y}_i)\right]$
    \item Lower deviance $\Rightarrow$ better fit
\end{itemize}

\section{Bayesian Model Comparison}
%-------------------------------

\begin{frame}{Bayes Factor}

\footnotesize

\textbf{Bayes Factor Definition}

\begin{itemize}

\item The Bayes factor compares two competing models

\item It is the ratio of marginal likelihoods

\end{itemize}

\vspace{0.2cm}

\begin{equation}
K = \frac{P(D|M_1)}{P(D|M_2)}
\end{equation}

\vspace{0.2cm}

\textbf{Marginal Likelihood}

\begin{equation}
P(D|M) = \int P(D|\theta, M)\, P(\theta|M)\, d\theta
\end{equation}

\vspace{0.2cm}

\textbf{Interpretation}

\begin{itemize}

\item $K > 1$ : Model $M_1$ preferred

\item $K < 1$ : Model $M_2$ preferred

\item Larger $K$ indicates stronger evidence

\end{itemize}

\end{frame}





\begin{frame}{Bayes Factor Approximation}

\footnotesize

\textbf{Monte Carlo Approximation}

\begin{equation}
P(D|M) \approx \frac{1}{N}\sum_{i=1}^{N} P(D|\theta_i , M)
\end{equation}

\vspace{0.2cm}

\textbf{Log Bayes Factor}

\begin{equation}
\log B = \log P(D|M_1) - \log P(D|M_2)
\end{equation}

\vspace{0.3cm}

\textbf{Note}

\begin{itemize}

\item Bayes factor was approximated using posterior samples

\item Monte Carlo estimation of marginal likelihoods

\end{itemize}

\end{frame}






\begin{frame}{Bayes Factor}

\footnotesize

\textbf{Bayes Factor Definition (Used in This Work)}

\[
\log B = \log L_g - \log L_p
\]

\begin{itemize}

\item $\log L_g$ : Gaussian model marginal likelihood

\item $\log L_p$ : Poisson model marginal likelihood

\end{itemize}

\vspace{0.2cm}

\textbf{Approximation Method}

\begin{itemize}

\item Bayes factor computed using marginal likelihoods

\item Monte Carlo approximation using posterior samples

\end{itemize}

\vspace{0.2cm}

\textbf{Computed Values}

\begin{itemize}

\item Log Bayes Factor = 2.8

\item Bayes Factor = 16.4

\end{itemize}

\end{frame}



\begin{frame}{Limitations of Bayes Factor}

\footnotesize

\textbf{Numerical Stability}

\begin{itemize}

\item Bayes factor computed using marginal likelihoods

\item Marginal likelihood estimated using MCMC posterior samples

\item This estimation can be numerically unstable

\item Results may depend on sampling quality

\end{itemize}

\vspace{0.3cm}

\textbf{Advanced Bayesian Model Comparison}

\begin{itemize}

\item Widely Applicable Information Criterion (WAIC)

\item Leave-One-Out Cross Validation (LOO)

\item These methods are more stable and robust

\end{itemize}

\vspace{0.2cm}

\textbf{In This Work}

\begin{itemize}

\item Bayes factor used as an approximate comparison

\item Final interpretation supported by posterior diagnostics and AIC/BIC

\end{itemize}

\end{frame}

\footnotesize

\textbf{Interpretation}

\begin{itemize}

\item $\log B > 0$ → (according to \textbf{Jeffery's scale}) Gaussian preferred

\item Moderate to strong evidence

\item Gaussian $\approx 16$ times more probable

\item Lower-degree polynomial favored

\end{itemize}

\vspace{0.3cm}

\textbf{Final Model}

\begin{itemize}

\item Degree 1 polynomial

\end{itemize}

\end{frame}
